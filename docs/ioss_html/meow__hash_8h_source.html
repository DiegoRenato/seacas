32  FAQ
33 
34  Q: What is it?
35 
36  A: Meow is a 128-bit Level 3 hash taking 128 bytes of seed. It operates
37  at very high speeds on x64 processors, and potentially other processors
38  that provide accelerated AES instructions.
39 
40  Q: What is it GOOD for?
41 
42  A: Quickly hashing any amount of data for comparison purposes such as
43  block deduplication or change detection. It is fast on all buffer
44  sizes, and can generally be used anywhere you need fast Level 3
45  hashing without worrying about how big or small the inputs tend to be.
46 
47  However, substantial speed improvements could be made over Meow
48  if you either a) know you are always hashing an exact, small number of bytes,
49  or b) can always supply a small number of bytes in a buffer padded to some
50  fixed multiple of 16.
51 
52  Q: What is it BAD for?
53 
54  A: Anything requiring Level 4 or Level 5 security guarantees (see
55  http://nohatcoder.dk/2019-05-19-1.html#level3). Also, note that
56  Meow is a new hash and has not had the extensive community
57  cryptanalysis necessary to ensure that it is not breakable down to
58  a lower level of hash, so you must do your due diligence in
59  deciding when and where to use Meow instead of a slower but
60  more extensively studied existing hash. We have tried to design
61  it to provide Level 3 security, but the possibility of the hash
62  being broken in the future always exists.
63 
64  Q: Why is it called the "Meow hash"?
65 
66  A: It is named after a character in Meow the Infinite
67  (https://meowtheinfinite.com)
68 
69  Q: Who wrote it?
70 
71  A: The final Meow Hash was created as a collaboration between
72  JACOB CHRISTIAN MUNCH-ANDERSEN (https://twitter.com/nohatcoder) and
73  CASEY MURATORI (https://caseymuratori.com). Casey wrote the original
74  implementation for use in processing large-footprint assets for the
75  game 1935 (https://molly1935.com). Jacob was the first to analyze
76  that implementation and determine the adversarial bit strength, which
77  was weaker than they would have liked.
78 
79  Following that, the two collaborated to figure out how the hash
80  could be strengthened without reducing Meow's 16 bytes/cycle
81  maximum theoretical throughput. Jacob created the hash candidates
82  and Casey did the performance validation. After a long and
83  exhaustive effort, Jacob found the unaligned aes/add/xor formulation
84  that forms the current Meow hash core.
85 
86  A number of valuable additions to Meow Hash were also contributed
87  by other great folks along the way:
88 
89  JEFF ROBERTS (https://radgametools.com) provided a super slick
90  way to handle the residual end-of-buffer bytes that dramatically
91  improved Meow's small hash performance.
92 
93  MARTINS MOZEIKO (https://matrins.ninja) ported Meow to ARM and
94  ANSI-C, and added the proper preprocessor dressing for clean
95  compilation on a variety of compiler configurations.
96 
97  FABIAN GIESEN (https://fgiesen.wordpress.com) analyzed many
98  performance oddities that came up during development, and
99  helped get the benchmarking working properly across a number
100  of platforms.
101 
102  ARAS PRANCKEVICIUS (https://aras-p.info) provided the allocation
103  shim for compilation on Mac OS X.

258 static meow_u128
259 MeowHash(void *Seed128Init, meow_umm Len, void *SourceInit)
260 {
261  meow_u128 xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7; // NOTE(casey): xmm0-xmm7 are the hash accumulation lanes
262  meow_u128 xmm8, xmm9, xmm10, xmm11, xmm12, xmm13, xmm14, xmm15; // NOTE(casey): xmm8-xmm15 hold values to be appended (residual, length)
263 
264  meow_u8 *rax = (meow_u8 *)SourceInit;
265  meow_u8 *rcx = (meow_u8 *)Seed128Init;
266 
267  //
268  // NOTE(casey): Seed the eight hash registers
269  //
270 
271  movdqu(xmm0, rcx + 0x00);
272  movdqu(xmm1, rcx + 0x10);
273  movdqu(xmm2, rcx + 0x20);
274  movdqu(xmm3, rcx + 0x30);
275 
276  movdqu(xmm4, rcx + 0x40);
277  movdqu(xmm5, rcx + 0x50);
278  movdqu(xmm6, rcx + 0x60);
279  movdqu(xmm7, rcx + 0x70);
280 
281  MEOW_DUMP_STATE("Seed", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
282 
283  //
284  // NOTE(casey): Hash all full 256-byte blocks
285  //
286 
287  meow_umm BlockCount = (Len >> 8);
288  if(BlockCount > MEOW_PREFETCH_LIMIT)
289  {
290  // NOTE(casey): For large input, modern Intel x64's can't hit full speed without prefetching, so we use this loop
291  while(BlockCount--)
292  {
293  prefetcht0(rax + MEOW_PREFETCH + 0x00);
294  prefetcht0(rax + MEOW_PREFETCH + 0x40);
295  prefetcht0(rax + MEOW_PREFETCH + 0x80);
296  prefetcht0(rax + MEOW_PREFETCH + 0xc0);
297 
298  MEOW_MIX(xmm0,xmm4,xmm6,xmm1,xmm2, rax + 0x00);
299  MEOW_MIX(xmm1,xmm5,xmm7,xmm2,xmm3, rax + 0x20);
300  MEOW_MIX(xmm2,xmm6,xmm0,xmm3,xmm4, rax + 0x40);
301  MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x60);
302  MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x80);
303  MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0xa0);
304  MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0xc0);
305  MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xe0);
306 
307  rax += 0x100;
308  }
309  }
310  else
311  {
312  // NOTE(casey): For small input, modern Intel x64's can't hit full speed _with_ prefetching (because of port pressure), so we use this loop.
313  while(BlockCount--)
314  {
315  MEOW_MIX(xmm0,xmm4,xmm6,xmm1,xmm2, rax + 0x00);
316  MEOW_MIX(xmm1,xmm5,xmm7,xmm2,xmm3, rax + 0x20);
317  MEOW_MIX(xmm2,xmm6,xmm0,xmm3,xmm4, rax + 0x40);
318  MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x60);
319  MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x80);
320  MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0xa0);
321  MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0xc0);
322  MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xe0);
323 
324  rax += 0x100;
325  }
326  }
327 
328  MEOW_DUMP_STATE("PostBlocks", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
329 
330  //
331  // NOTE(casey): Load any less-than-32-byte residual
332  //
333 
334  pxor_clear(xmm9, xmm9);
335  pxor_clear(xmm11, xmm11);
336 
337  //
338  // TODO(casey): I need to put more thought into how the end-of-buffer stuff is actually working out here,
339  // because I _think_ it may be possible to remove the first branch (on Len8) and let the mask zero out the
340  // result, but it would take a little thought to make sure it couldn't read off the end of the buffer due
341  // to the & 0xf on the align computation.
342  //
343 
344  // NOTE(casey): First, we have to load the part that is _not_ 16-byte aligned
345  meow_u8 *Last = (meow_u8 *)SourceInit + (Len & ~0xf);
346  int unsigned Len8 = (Len & 0xf);
347  if(Len8)
348  {
349  // NOTE(casey): Load the mask early
350  movdqu(xmm8, &MeowMaskLen[0x10 - Len8]);
351 
352  meow_u8 *LastOk = (meow_u8*)((((meow_umm)(((meow_u8 *)SourceInit)+Len - 1)) | (MEOW_PAGESIZE - 1)) - 16);
353  int Align = (Last > LastOk) ? ((int)(meow_umm)Last) & 0xf : 0;
354  movdqu(xmm10, &MeowShiftAdjust[Align]);
355  movdqu(xmm9, Last - Align);
356  pshufb(xmm9, xmm10);
357 
358  // NOTE(jeffr): and off the extra bytes
359  pand(xmm9, xmm8);
360  }
361 
362  // NOTE(casey): Next, we have to load the part that _is_ 16-byte aligned
363  if(Len & 0x10)
364  {
365  xmm11 = xmm9;
366  movdqu(xmm9, Last - 0x10);
367  }
368 
369  //
370  // NOTE(casey): Construct the residual and length injests
371  //
372 
373  xmm8 = xmm9;
374  xmm10 = xmm9;
375  palignr(xmm8, xmm11, 15);
376  palignr(xmm10, xmm11, 1);
377 
378  // NOTE(casey): We have room for a 128-bit nonce and a 64-bit none here, but
379  // the decision was made to leave them zero'd so as not to confuse people
380  // about hwo to use them or what security implications they had.
381  pxor_clear(xmm12, xmm12);
382  pxor_clear(xmm13, xmm13);
383  pxor_clear(xmm14, xmm14);
384  movq(xmm15, Len);
385  palignr(xmm12, xmm15, 15);
386  palignr(xmm14, xmm15, 1);
387 
388  MEOW_DUMP_STATE("Residuals", xmm8, xmm9, xmm10, xmm11, xmm12, xmm13, xmm14, xmm15, 0);
389 
390  // NOTE(casey): To maintain the mix-down pattern, we always Meow Mix the less-than-32-byte residual, even if it was empty
391  MEOW_MIX_REG(xmm0, xmm4, xmm6, xmm1, xmm2, xmm8, xmm9, xmm10, xmm11);
392 
393  // NOTE(casey): Append the length, to avoid problems with our 32-byte padding
394  MEOW_MIX_REG(xmm1, xmm5, xmm7, xmm2, xmm3, xmm12, xmm13, xmm14, xmm15);
395 
396  MEOW_DUMP_STATE("PostAppend", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
397 
398  //
399  // NOTE(casey): Hash all full 32-byte blocks
400  //
401  int unsigned LaneCount = (Len >> 5) & 0x7;
402  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm2,xmm6,xmm0,xmm3,xmm4, rax + 0x00); --LaneCount;
403  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x20); --LaneCount;
404  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x40); --LaneCount;
405  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0x60); --LaneCount;
406  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0x80); --LaneCount;
407  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xa0); --LaneCount;
408  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm0,xmm4,xmm6,xmm1,xmm2, rax + 0xc0); --LaneCount;
409 
410  //
411  // NOTE(casey): Mix the eight lanes down to one 128-bit hash
412  //
413 
414  MixDown:
415 
416  MEOW_DUMP_STATE("PostLanes", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
417 
418  MEOW_SHUFFLE(xmm0, xmm1, xmm2, xmm4, xmm5, xmm6);
419  MEOW_SHUFFLE(xmm1, xmm2, xmm3, xmm5, xmm6, xmm7);
420  MEOW_SHUFFLE(xmm2, xmm3, xmm4, xmm6, xmm7, xmm0);
421  MEOW_SHUFFLE(xmm3, xmm4, xmm5, xmm7, xmm0, xmm1);
422  MEOW_SHUFFLE(xmm4, xmm5, xmm6, xmm0, xmm1, xmm2);
423  MEOW_SHUFFLE(xmm5, xmm6, xmm7, xmm1, xmm2, xmm3);
424  MEOW_SHUFFLE(xmm6, xmm7, xmm0, xmm2, xmm3, xmm4);
425  MEOW_SHUFFLE(xmm7, xmm0, xmm1, xmm3, xmm4, xmm5);
426  MEOW_SHUFFLE(xmm0, xmm1, xmm2, xmm4, xmm5, xmm6);
427  MEOW_SHUFFLE(xmm1, xmm2, xmm3, xmm5, xmm6, xmm7);
428  MEOW_SHUFFLE(xmm2, xmm3, xmm4, xmm6, xmm7, xmm0);
429  MEOW_SHUFFLE(xmm3, xmm4, xmm5, xmm7, xmm0, xmm1);
430 
431  MEOW_DUMP_STATE("PostMix", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
432 
433  paddq(xmm0, xmm2);
434  paddq(xmm1, xmm3);
435  paddq(xmm4, xmm6);
436  paddq(xmm5, xmm7);
437  pxor(xmm0, xmm1);
438  pxor(xmm4, xmm5);
439  paddq(xmm0, xmm4);
440 
441  MEOW_DUMP_STATE("PostFold", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
442 
443  return(xmm0);
444 }

483 {

485  meow_u128 xmm1 = State->xmm1;
486  meow_u128 xmm2 = State->xmm2;
487  meow_u128 xmm3 = State->xmm3;
488  meow_u128 xmm4 = State->xmm4;
489  meow_u128 xmm5 = State->xmm5;
490  meow_u128 xmm6 = State->xmm6;
491  meow_u128 xmm7 = State->xmm7;
492

494  {

497  prefetcht0(rax + MEOW_PREFETCH + 0x00);
498  prefetcht0(rax + MEOW_PREFETCH + 0x40);

505  MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x60);
506  MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x80);
507  MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0xa0);
508  MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0xc0);
509  MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xe0);
510 
511  rax += 0x100;
512  }
513  }
514  else
515  {
516  while(BlockCount--)
517  {
518  MEOW_MIX(xmm0,xmm4,xmm6,xmm1,xmm2, rax + 0x00);

520  MEOW_MIX(xmm2,xmm6,xmm0,xmm3,xmm4, rax + 0x40);
521  MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x60);
522  MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x80);
523  MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0xa0);
524  MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0xc0);
525  MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xe0);
526 
527  rax += 0x100;
528  }
529  }

531  State->xmm0 = xmm0;
532  State->xmm1 = xmm1;
533  State->xmm2 = xmm2;
534  State->xmm3 = xmm3;

541 static void
542 MeowAbsorb(meow_state *State, meow_umm Len, void *SourceInit)
543 {
544  State->TotalLengthInBytes += Len;
545  meow_u8 *Source = (meow_u8 *)SourceInit;
546 
547  // NOTE(casey): Handle any buffered residual
548  if(State->BufferLen)
549  {
550  int unsigned Fill = (sizeof(State->Buffer) - State->BufferLen);
551  if(Fill > Len)
552  {
553  Fill = (int unsigned)Len;
554  }
555 
556  Len -= Fill;
557  while(Fill--)
558  {
559  State->Buffer[State->BufferLen++] = *Source++;
560  }
561 
562  if(State->BufferLen == sizeof(State->Buffer))
563  {
564  MeowAbsorbBlocks(State, 1, State->Buffer);
565  State->BufferLen = 0;
566  }
567  }
568 
569  // NOTE(casey): Handle any full blocks
570  meow_u64 BlockCount = (Len >> 8);
571  meow_u64 Advance = (BlockCount << 8);
572  MeowAbsorbBlocks(State, BlockCount, Source);
573 
574  Len -= Advance;
575  Source += Advance;
576 
577  // NOTE(casey): Store residual
578  while(Len--)
579  {
580  State->Buffer[State->BufferLen++] = *Source++;
581  }
582 }

584 static meow_u128
585 MeowEnd(meow_state *State, meow_u8 *Store128)
586 {
587  meow_umm Len = State->TotalLengthInBytes;
588 
589  meow_u128 xmm0 = State->xmm0;
590  meow_u128 xmm1 = State->xmm1;
591  meow_u128 xmm2 = State->xmm2;
592  meow_u128 xmm3 = State->xmm3;
593  meow_u128 xmm4 = State->xmm4;
594  meow_u128 xmm5 = State->xmm5;
595  meow_u128 xmm6 = State->xmm6;
596  meow_u128 xmm7 = State->xmm7;
597 
598  meow_u128 xmm8, xmm9, xmm10, xmm11, xmm12, xmm13, xmm14, xmm15;
599 
600  meow_u8 *rax = State->Buffer;
601 
602  pxor_clear(xmm9, xmm9);
603  pxor_clear(xmm11, xmm11);
604 
605  meow_u8 *Last = (meow_u8 *)rax + (Len & 0xf0);
606  int unsigned Len8 = (Len & 0xf);
607  if(Len8)
608  {
609  movdqu(xmm8, &MeowMaskLen[0x10 - Len8]);
610  movdqu(xmm9, Last);
611  pand(xmm9, xmm8);
612  }
613 
614  if(Len & 0x10)
615  {
616  xmm11 = xmm9;
617  movdqu(xmm9, Last - 0x10);
618  }
619 
620  xmm8 = xmm9;
621  xmm10 = xmm9;
622  palignr(xmm8, xmm11, 15);
623  palignr(xmm10, xmm11, 1);
624 
625  pxor_clear(xmm12, xmm12);
626  pxor_clear(xmm13, xmm13);
627  pxor_clear(xmm14, xmm14);
628  movq(xmm15, Len);
629  palignr(xmm12, xmm15, 15);
630  palignr(xmm14, xmm15, 1);
631 
632  MEOW_DUMP_STATE("PostBlocks", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
633  MEOW_DUMP_STATE("Residuals", xmm8, xmm9, xmm10, xmm11, xmm12, xmm13, xmm14, xmm15, 0);
634 
635  // NOTE(casey): To maintain the mix-down pattern, we always Meow Mix the less-than-32-byte residual, even if it was empty
636  MEOW_MIX_REG(xmm0, xmm4, xmm6, xmm1, xmm2, xmm8, xmm9, xmm10, xmm11);
637 
638  // NOTE(casey): Append the length, to avoid problems with our 32-byte padding
639  MEOW_MIX_REG(xmm1, xmm5, xmm7, xmm2, xmm3, xmm12, xmm13, xmm14, xmm15);
640 
641  MEOW_DUMP_STATE("PostAppend", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
642 
643  //
644  // NOTE(casey): Hash all full 32-byte blocks
645  //
646  int unsigned LaneCount = (Len >> 5) & 0x7;
647  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm2,xmm6,xmm0,xmm3,xmm4, rax + 0x00); --LaneCount;
648  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm3,xmm7,xmm1,xmm4,xmm5, rax + 0x20); --LaneCount;
649  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm4,xmm0,xmm2,xmm5,xmm6, rax + 0x40); --LaneCount;
650  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm5,xmm1,xmm3,xmm6,xmm7, rax + 0x60); --LaneCount;
651  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm6,xmm2,xmm4,xmm7,xmm0, rax + 0x80); --LaneCount;
652  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm7,xmm3,xmm5,xmm0,xmm1, rax + 0xa0); --LaneCount;
653  if(LaneCount == 0) goto MixDown; MEOW_MIX(xmm0,xmm4,xmm6,xmm1,xmm2, rax + 0xc0); --LaneCount;
654 
655  //
656  // NOTE(casey): Mix the eight lanes down to one 128-bit hash
657  //
658 
659  MixDown:
660 
661  MEOW_DUMP_STATE("PostLanes", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
662 
663  MEOW_SHUFFLE(xmm0, xmm1, xmm2, xmm4, xmm5, xmm6);
664  MEOW_SHUFFLE(xmm1, xmm2, xmm3, xmm5, xmm6, xmm7);
665  MEOW_SHUFFLE(xmm2, xmm3, xmm4, xmm6, xmm7, xmm0);
666  MEOW_SHUFFLE(xmm3, xmm4, xmm5, xmm7, xmm0, xmm1);
667  MEOW_SHUFFLE(xmm4, xmm5, xmm6, xmm0, xmm1, xmm2);
668  MEOW_SHUFFLE(xmm5, xmm6, xmm7, xmm1, xmm2, xmm3);
669  MEOW_SHUFFLE(xmm6, xmm7, xmm0, xmm2, xmm3, xmm4);
670  MEOW_SHUFFLE(xmm7, xmm0, xmm1, xmm3, xmm4, xmm5);
671  MEOW_SHUFFLE(xmm0, xmm1, xmm2, xmm4, xmm5, xmm6);
672  MEOW_SHUFFLE(xmm1, xmm2, xmm3, xmm5, xmm6, xmm7);
673  MEOW_SHUFFLE(xmm2, xmm3, xmm4, xmm6, xmm7, xmm0);
674  MEOW_SHUFFLE(xmm3, xmm4, xmm5, xmm7, xmm0, xmm1);
675 
676  MEOW_DUMP_STATE("PostMix", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
677 
678  if(Store128)
679  {
680  movdqu_mem(Store128 + 0x00, xmm0);
681  movdqu_mem(Store128 + 0x10, xmm1);
682  movdqu_mem(Store128 + 0x20, xmm2);
683  movdqu_mem(Store128 + 0x30, xmm3);
684  movdqu_mem(Store128 + 0x40, xmm4);
685  movdqu_mem(Store128 + 0x50, xmm5);
686  movdqu_mem(Store128 + 0x60, xmm6);
687  movdqu_mem(Store128 + 0x70, xmm7);
688  }
689 
690  paddq(xmm0, xmm2);
691  paddq(xmm1, xmm3);
692  paddq(xmm4, xmm6);
693  paddq(xmm5, xmm7);
694  pxor(xmm0, xmm1);
695  pxor(xmm4, xmm5);
696  paddq(xmm0, xmm4);
697 
698  MEOW_DUMP_STATE("PostFold", xmm0, xmm1, xmm2, xmm3, xmm4, xmm5, xmm6, xmm7, 0);
699 
700  return(xmm0);
701 }
