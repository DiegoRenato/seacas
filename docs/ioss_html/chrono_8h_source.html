43 {

45  {

48 
49  dispatcher(std::time_t t) : time_(t) {}
50 
51  bool run()
52  {
53  using namespace fmt::internal;
54  return handle(localtime_r(&time_, &tm_));
55  }
56 
57  bool handle(std::tm *tm) { return tm != nullptr; }
58 
59  bool handle(internal::null<>)
60  {
61  using namespace fmt::internal;
62  return fallback(localtime_s(&tm_, &time_));
63  }
64 
65  bool fallback(int res) { return res == 0; }
66 
67 #if !FMT_MSC_VER

70  using namespace fmt::internal;

141 template <typename Char> struct formatter<std::tm, Char>
142 {
143  template <typename ParseContext> auto parse(ParseContext &ctx) -> decltype(ctx.begin())
144  {
145  auto it = ctx.begin();
146  if (it != ctx.end() && *it == ':')
147  ++it;
148  auto end = it;
149  while (end != ctx.end() && *end != '}')
150  ++end;
151  tm_format.reserve(internal::to_unsigned(end - it + 1));
152  tm_format.append(it, end);
153  tm_format.push_back('\0');
154  return end;
155  }
156 
157  template <typename FormatContext>
158  auto format(const std::tm &tm, FormatContext &ctx) -> decltype(ctx.out())
159  {
160  basic_memory_buffer<Char> buf;
161  std::size_t start = buf.size();
162  for (;;) {
163  std::size_t size = buf.capacity() - start;
164  std::size_t count = internal::strftime(&buf[start], size, &tm_format[0], &tm);
165  if (count != 0) {
166  buf.resize(start + count);
167  break;
168  }
169  if (size >= tm_format.size() * 256) {
170  // If the buffer is 256 times larger than the format string, assume
171  // that `strftime` gives an empty result. There doesn't seem to be a
172  // better way to distinguish the two cases:
173  // https://github.com/fmtlib/fmt/issues/367
174  break;
175  }
176  const std::size_t MIN_GROWTH = 10;
177  buf.reserve(buf.capacity() + (size > MIN_GROWTH ? size : MIN_GROWTH));
178  }
179  return std::copy(buf.begin(), buf.end(), ctx.out());
180  }
181 
182  basic_memory_buffer<Char> tm_format;
183 };

213  // Parses a put_time-like format string and invokes handler actions.
214  template <typename Char, typename Handler>
215  FMT_CONSTEXPR const Char *parse_chrono_format(const Char *begin, const Char *end,
216  Handler &&handler)
217  {
218  auto ptr = begin;
219  while (ptr != end) {
220  auto c = *ptr;
221  if (c == '}')
222  break;
223  if (c != '%') {
224  ++ptr;
225  continue;
226  }
227  if (begin != ptr)
228  handler.on_text(begin, ptr);
229  ++ptr; // consume '%'
230  if (ptr == end)
231  FMT_THROW(format_error("invalid format"));
232  c = *ptr++;
233  switch (c) {
234  case '%': handler.on_text(ptr - 1, ptr); break;
235  case 'n': {
236  const char newline[] = "\n";
237  handler.on_text(newline, newline + 1);
238  break;
239  }
240  case 't': {
241  const char tab[] = "\t";
242  handler.on_text(tab, tab + 1);
243  break;
244  }
245  // Day of the week:
246  case 'a': handler.on_abbr_weekday(); break;
247  case 'A': handler.on_full_weekday(); break;
248  case 'w': handler.on_dec0_weekday(numeric_system::standard); break;
249  case 'u': handler.on_dec1_weekday(numeric_system::standard); break;
250  // Month:
251  case 'b': handler.on_abbr_month(); break;
252  case 'B': handler.on_full_month(); break;
253  // Hour, minute, second:
254  case 'H': handler.on_24_hour(numeric_system::standard); break;
255  case 'I': handler.on_12_hour(numeric_system::standard); break;
256  case 'M': handler.on_minute(numeric_system::standard); break;
257  case 'S': handler.on_second(numeric_system::standard); break;
258  // Other:
259  case 'c': handler.on_datetime(numeric_system::standard); break;
260  case 'x': handler.on_loc_date(numeric_system::standard); break;
261  case 'X': handler.on_loc_time(numeric_system::standard); break;
262  case 'D': handler.on_us_date(); break;
263  case 'F': handler.on_iso_date(); break;
264  case 'r': handler.on_12_hour_time(); break;
265  case 'R': handler.on_24_hour_time(); break;
266  case 'T': handler.on_iso_time(); break;
267  case 'p': handler.on_am_pm(); break;
268  case 'Q': handler.on_duration_value(); break;
269  case 'q': handler.on_duration_unit(); break;
270  case 'z': handler.on_utc_offset(); break;
271  case 'Z': handler.on_tz_name(); break;
272  // Alternative representation:
273  case 'E': {
274  if (ptr == end)
275  FMT_THROW(format_error("invalid format"));
276  c = *ptr++;
277  switch (c) {
278  case 'c': handler.on_datetime(numeric_system::alternative); break;
279  case 'x': handler.on_loc_date(numeric_system::alternative); break;
280  case 'X': handler.on_loc_time(numeric_system::alternative); break;
281  default: FMT_THROW(format_error("invalid format"));
282  }
283  break;
284  }
285  case 'O':
286  if (ptr == end)
287  FMT_THROW(format_error("invalid format"));
288  c = *ptr++;
289  switch (c) {
290  case 'w': handler.on_dec0_weekday(numeric_system::alternative); break;
291  case 'u': handler.on_dec1_weekday(numeric_system::alternative); break;
292  case 'H': handler.on_24_hour(numeric_system::alternative); break;
293  case 'I': handler.on_12_hour(numeric_system::alternative); break;
294  case 'M': handler.on_minute(numeric_system::alternative); break;
295  case 'S': handler.on_second(numeric_system::alternative); break;
296  default: FMT_THROW(format_error("invalid format"));
297  }
298  break;
299  default: FMT_THROW(format_error("invalid format"));
300  }
301  begin = ptr;
302  }
303  if (begin != ptr)
304  handler.on_text(begin, ptr);
305  return ptr;
306  }

409  template <typename Rep, typename Period, FMT_ENABLE_IF(std::is_integral<Rep>::value)>
410  inline std::chrono::duration<Rep, std::milli>
411  get_milliseconds(std::chrono::duration<Rep, Period> d)
412  {
413  // this may overflow and/or the result may not fit in the
414  // target type.
415 #if FMT_SAFE_DURATION_CAST
416  using CommonSecondsType = typename std::common_type<decltype(d), std::chrono::seconds>::type;
417  const auto d_as_common = fmt_safe_duration_cast<CommonSecondsType>(d);
418  const auto d_as_whole_seconds = fmt_safe_duration_cast<std::chrono::seconds>(d_as_common);
419  // this conversion should be nonproblematic
420  const auto diff = d_as_common - d_as_whole_seconds;
421  const auto ms = fmt_safe_duration_cast<std::chrono::duration<Rep, std::milli>>(diff);
422  return ms;
423 #else
424  auto s = std::chrono::duration_cast<std::chrono::seconds>(d);
425  return std::chrono::duration_cast<std::chrono::milliseconds>(d - s);
426 #endif
427  }

429  template <typename Rep, typename Period, FMT_ENABLE_IF(std::is_floating_point<Rep>::value)>
430  inline std::chrono::duration<Rep, std::milli>
431  get_milliseconds(std::chrono::duration<Rep, Period> d)
432  {
433  using common_type = typename std::common_type<Rep, std::intmax_t>::type;
434  auto ms = mod(d.count() * static_cast<common_type>(Period::num) /

437  return std::chrono::duration<Rep, std::milli>(static_cast<Rep>(ms));

458  template <typename FormatContext, typename OutputIt, typename Rep, typename Period>
459  struct chrono_formatter
460  {
461  FormatContext &context;
462  OutputIt out;
463  int precision;
464  // rep is unsigned to avoid overflow.
465  using rep = conditional_t<std::is_integral<Rep>::value && sizeof(Rep) < sizeof(int), unsigned,
466  typename make_unsigned_or_unchanged<Rep>::type>;
467  rep val;
468  using seconds = std::chrono::duration<rep>;
469  seconds s;
470  using milliseconds = std::chrono::duration<rep, std::milli>;
471  bool negative;
472 
473  using char_type = typename FormatContext::char_type;
474 
475  explicit chrono_formatter(FormatContext &ctx, OutputIt o, std::chrono::duration<Rep, Period> d)
476  : context(ctx), out(o), val(d.count()), negative(false)
477  {
478  if (d.count() < 0) {
479  val = 0 - val;
480  negative = true;
481  }

484  // target type.
485 #if FMT_SAFE_DURATION_CAST
486  // might need checked conversion (rep!=Rep)
487  auto tmpval = std::chrono::duration<rep, Period>(val);
488  s = fmt_safe_duration_cast<seconds>(tmpval);
489 #else
490  s = std::chrono::duration_cast<seconds>(std::chrono::duration<rep, Period>(val));

759  FMT_CONSTEXPR parse_range do_parse(basic_parse_context<Char> &ctx)
760  {
761  auto begin = ctx.begin(), end = ctx.end();
762  if (begin == end || *begin == '}')
763  return {begin, begin};
764  spec_handler handler{*this, ctx, format_str};
765  begin = internal::parse_align(begin, end, handler);
766  if (begin == end)
767  return {begin, begin};
768  begin = internal::parse_width(begin, end, handler);
769  if (begin == end)
770  return {begin, begin};
771  if (*begin == '.') {
772  if (std::is_floating_point<Rep>::value)
773  begin = internal::parse_precision(begin, end, handler);
774  else
775  handler.on_error("precision not allowed for this argument type");
776  }
777  end = parse_chrono_format(begin, end, internal::chrono_format_checker());
778  return {begin, end};
779  }

783

785  {
786  auto range = do_parse(ctx);
787  format_str =
788  basic_string_view<Char>(&*range.begin, internal::to_unsigned(range.end - range.begin));
789  return range.end;
790  }

792  template <typename FormatContext>
793  auto format(const duration &d, FormatContext &ctx) -> decltype(ctx.out())
794  {
795  auto begin = format_str.begin(), end = format_str.end();
796  // As a possible future optimization, we could avoid extra copying if width
797  // is not specified.
798  basic_memory_buffer<Char> buf;
799  auto out = std::back_inserter(buf);
800  using range = internal::output_range<decltype(ctx.out()), Char>;
801  internal::basic_writer<range> w(range(ctx.out()));
802  internal::handle_dynamic_spec<internal::width_checker>(specs.width, width_ref, ctx,
803  format_str.begin());
804  internal::handle_dynamic_spec<internal::precision_checker>(precision, precision_ref, ctx,
805  format_str.begin());
806  if (begin == end || *begin == '}') {
807  out = internal::format_chrono_duration_value(out, d.count(), precision);
808  internal::format_chrono_duration_unit<Period>(out);
809  }
810  else {
811  internal::chrono_formatter<FormatContext, decltype(out), Rep, Period> f(ctx, out, d);
812  f.precision = precision;
813  parse_chrono_format(begin, end, f);
814  }
815  w.write(buf.data(), buf.size(), specs);
816  return w.out();
817  }
