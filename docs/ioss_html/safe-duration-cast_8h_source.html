155  /**
156  * safe duration cast between integral durations
157  */
158  template <typename To, typename FromRep, typename FromPeriod,
159  FMT_ENABLE_IF(std::is_integral<FromRep>::value),
160  FMT_ENABLE_IF(std::is_integral<typename To::rep>::value)>
161  To safe_duration_cast(std::chrono::duration<FromRep, FromPeriod> from, int &ec)
162  {
163  using From = std::chrono::duration<FromRep, FromPeriod>;
164  ec = 0;
165  // the basic idea is that we need to convert from count() in the from type
166  // to count() in the To type, by multiplying it with this:
167  using Factor = std::ratio_divide<typename From::period, typename To::period>;
168 
169  static_assert(Factor::num > 0, "num must be positive");
170  static_assert(Factor::den > 0, "den must be positive");
171 
172  // the conversion is like this: multiply from.count() with Factor::num
173  // /Factor::den and convert it to To::rep, all this without
174  // overflow/underflow. let's start by finding a suitable type that can hold
175  // both To, From and Factor::num
176  using IntermediateRep = typename std::common_type<typename From::rep, typename To::rep,
177  decltype(Factor::num)>::type;
178 
179  // safe conversion to IntermediateRep
180  IntermediateRep count = lossless_integral_conversion<IntermediateRep>(from.count(), ec);
181  if (ec) {
182  return {};
183  }
184  // multiply with Factor::num without overflow or underflow
185  if (Factor::num != 1) {
186  constexpr auto max1 = std::numeric_limits<IntermediateRep>::max() / Factor::num;
187  if (count > max1) {
188  ec = 1;
189  return {};
190  }
191  constexpr auto min1 = std::numeric_limits<IntermediateRep>::min() / Factor::num;
192  if (count < min1) {
193  ec = 1;
194  return {};
195  }
196  count *= Factor::num;
197  }
198 
199  // this can't go wrong, right? den>0 is checked earlier.
200  if (Factor::den != 1) {
201  count /= Factor::den;
202  }
203  // convert to the to type, safely
204  using ToRep = typename To::rep;
205  const ToRep tocount = lossless_integral_conversion<ToRep>(count, ec);
206  if (ec) {
207  return {};
208  }
209  return To{tocount};
210  }

227  // it.

270  }

273  if (Factor::den != 1) {
274  using common_t = typename std::common_type<IntermediateRep, intmax_t>::type;
275  count /= static_cast<common_t>(Factor::den);
