245 /**
246  An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
247  subset of the API. ``fmt::basic_string_view`` is used for format strings even
248  if ``std::string_view`` is available to prevent issues when a library is
249  compiled with a different ``-std`` option than the client code (which is not
250  recommended).
251  */
252 template <typename Char> class basic_string_view
253 {
254 private:
255  const Char *data_;
256  size_t size_;
257 
258 public:
259  using char_type = Char;
260  using iterator = const Char *;
261 
262  FMT_CONSTEXPR basic_string_view() FMT_NOEXCEPT : data_(nullptr), size_(0) {}
263 
264  /** Constructs a string reference object from a C string and a size. */
265  FMT_CONSTEXPR basic_string_view(const Char *s, size_t count) FMT_NOEXCEPT : data_(s), size_(count)
266  {
267  }
268 
269  /**
270  \rst
271  Constructs a string reference object from a C string computing
272  the size with ``std::char_traits<Char>::length``.
273  \endrst
274  */
275  basic_string_view(const Char *s) : data_(s), size_(std::char_traits<Char>::length(s)) {}
276 
277  /** Constructs a string reference from a ``std::basic_string`` object. */
278  template <typename Alloc>
279  FMT_CONSTEXPR basic_string_view(const std::basic_string<Char, Alloc> &s) FMT_NOEXCEPT
280  : data_(s.data()),
281  size_(s.size())
282  {
283  }
284 
285  template <typename S, FMT_ENABLE_IF(std::is_same<S, internal::std_string_view<Char>>::value)>
286  FMT_CONSTEXPR basic_string_view(S s) FMT_NOEXCEPT : data_(s.data()), size_(s.size())
287  {
288  }
289 
290  /** Returns a pointer to the string data. */
291  FMT_CONSTEXPR const Char *data() const { return data_; }
292 
293  /** Returns the string size. */
294  FMT_CONSTEXPR size_t size() const { return size_; }
295 
296  FMT_CONSTEXPR iterator begin() const { return data_; }
297  FMT_CONSTEXPR iterator end() const { return data_ + size_; }
298 
299  FMT_CONSTEXPR void remove_prefix(size_t n)
300  {
301  data_ += n;
302  size_ -= n;
303  }
304 
305  // Lexicographically compare this string reference to other.
306  int compare(basic_string_view other) const
307  {
308  size_t str_size = size_ < other.size_ ? size_ : other.size_;
309  int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
310  if (result == 0)
311  result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
312  return result;
313  }

315  friend bool operator==(basic_string_view lhs, basic_string_view rhs)
316  {
317  return lhs.compare(rhs) == 0;
318  }

320  {
321  return lhs.compare(rhs) != 0;
322  }
323  friend bool operator<(basic_string_view lhs, basic_string_view rhs)
324  {
325  return lhs.compare(rhs) < 0;
326  }
327  friend bool operator<=(basic_string_view lhs, basic_string_view rhs)
328  {

545  /** A contiguous memory buffer with an optional growing ability. */
546  template <typename T> class buffer
547  {
548  private:
549  buffer(const buffer &) = delete;
550  void operator=(const buffer &) = delete;
551 
552  T * ptr_;
553  std::size_t size_;
554  std::size_t capacity_;
555 
556  protected:
557  // Don't initialize ptr_ since it is not accessed to save a few cycles.
558  buffer(std::size_t sz) FMT_NOEXCEPT : size_(sz), capacity_(sz) {}
559 
560  buffer(T *p = nullptr, std::size_t sz = 0, std::size_t cap = 0) FMT_NOEXCEPT : ptr_(p),
561  size_(sz),
562  capacity_(cap)
563  {
564  }
565 
566  /** Sets the buffer data and capacity. */
567  void set(T *buf_data, std::size_t buf_capacity) FMT_NOEXCEPT
568  {
569  ptr_ = buf_data;
570  capacity_ = buf_capacity;
571  }
572 
573  /** Increases the buffer capacity to hold at least *capacity* elements. */
574  virtual void grow(std::size_t capacity) = 0;
575 
576  public:
577  using value_type = T;
578  using const_reference = const T &;
579 
580  virtual ~buffer() {}
581 
582  T *begin() FMT_NOEXCEPT { return ptr_; }
583  T *end() FMT_NOEXCEPT { return ptr_ + size_; }
584 
585  /** Returns the size of this buffer. */
586  std::size_t size() const FMT_NOEXCEPT { return size_; }
587 
588  /** Returns the capacity of this buffer. */
589  std::size_t capacity() const FMT_NOEXCEPT { return capacity_; }
590 
591  /** Returns a pointer to the buffer data. */
592  T *data() FMT_NOEXCEPT { return ptr_; }
593 
594  /** Returns a pointer to the buffer data. */
595  const T *data() const FMT_NOEXCEPT { return ptr_; }
596 
597  /**
598  Resizes the buffer. If T is a POD type new elements may not be initialized.
599  */
600  void resize(std::size_t new_size)
601  {
602  reserve(new_size);
603  size_ = new_size;
604  }
605 
606  /** Clears this buffer. */
607  void clear() { size_ = 0; }
608 
609  /** Reserves space to store at least *capacity* elements. */
610  void reserve(std::size_t new_capacity)
611  {
612  if (new_capacity > capacity_)
613  grow(new_capacity);
614  }
615 
616  void push_back(const T &value)
617  {
618  reserve(size_ + 1);
619  ptr_[size_++] = value;
620  }
621 
622  /** Appends data to the end of the buffer. */
623  template <typename U> void append(const U *begin, const U *end);

626  const T &operator[](std::size_t index) const { return ptr_[index]; }

968 /**
969  \rst
970  Visits an argument dispatching to the appropriate visit method based on
971  the argument type. For example, if the argument type is ``double`` then
972  ``vis(value)`` will be called with the value of type ``double``.
973  \endrst
974  */
975 template <typename Visitor, typename Context>
976 FMT_CONSTEXPR auto visit_format_arg(Visitor &&vis, const basic_format_arg<Context> &arg)
977  -> decltype(vis(0))
978 {
979  using char_type = typename Context::char_type;
980  switch (arg.type_) {
981  case internal::none_type: break;
982  case internal::named_arg_type: FMT_ASSERT(false, "invalid argument type"); break;
983  case internal::int_type: return vis(arg.value_.int_value);
984  case internal::uint_type: return vis(arg.value_.uint_value);
985  case internal::long_long_type: return vis(arg.value_.long_long_value);
986  case internal::ulong_long_type: return vis(arg.value_.ulong_long_value);
987  case internal::bool_type: return vis(arg.value_.bool_value);
988  case internal::char_type: return vis(arg.value_.char_value);
989  case internal::double_type: return vis(arg.value_.double_value);
990  case internal::long_double_type: return vis(arg.value_.long_double_value);
991  case internal::cstring_type: return vis(arg.value_.string.data);
992  case internal::string_type:
993  return vis(basic_string_view<char_type>(arg.value_.string.data, arg.value_.string.size));
994  case internal::pointer_type: return vis(arg.value_.pointer);
995  case internal::custom_type:
996  return vis(typename basic_format_arg<Context>::handle(arg.value_.custom));
997  }
998  return vis(monostate());
999 }

1185 {
1186 public:
1187  using size_type = int;
1188  using format_arg = basic_format_arg<Context>;
1189 
1190 private:
1191  // To reduce compiled code size per formatting function call, types of first
1192  // max_packed_args arguments are passed in the types_ field.
1193  unsigned long long types_;
1194  union {
1195  // If the number of arguments is less than max_packed_args, the argument
1196  // values are stored in values_, otherwise they are stored in args_.
1197  // This is done to reduce compiled code size as storing larger objects
1198  // may require more code (at least on x86-64) even if the same amount of
1199  // data is actually copied to stack. It saves ~10% on the bloat test.
1200  const internal::value<Context> *values_;
1201  const format_arg * args_;
1202  };
1203 
1204  bool is_packed() const { return (types_ & internal::is_unpacked_bit) == 0; }
1205

1207  {
1208  int shift = index * 4;
1209  return static_cast<internal::type>((types_ & (0xfull << shift)) >> shift);
1210  }

1215  void set_data(const format_arg *args) { args_ = args; }

1218  {
1219  format_arg arg;
1220  if (!is_packed()) {
1221  auto num_args = max_size();
1222  if (index < num_args)
1223  arg = args_[index];
1224  return arg;
1225  }
1226  if (index > internal::max_packed_args)
1227  return arg;
1228  arg.type_ = type(index);
1229  if (arg.type_ == internal::none_type)
1230  return arg;
1231  internal::value<Context> &val = arg.value_;
1232  val = values_[index];
1233  return arg;
