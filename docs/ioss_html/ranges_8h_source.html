248  auto format(const TupleT &values, FormatContext &ctx) -> decltype(ctx.out())
249  {
250  auto out = ctx.out();

252  internal::copy(formatting.prefix, out);

282  template <typename FormatContext>
283  typename FormatContext::iterator format(const RangeT &values, FormatContext &ctx)
284  {
285  auto out = internal::copy(formatting.prefix, ctx.out());
286  std::size_t i = 0;
287  for (auto it = values.begin(), end = values.end(); it != end; ++it) {
288  if (i > 0) {
289  if (formatting.add_prepostfix_space)
290  *out++ = ' ';
291  out = internal::copy(formatting.delimiter, out);
292  }
293  out = format_to(
294  out, internal::format_str_quoted((formatting.add_delimiter_spaces && i > 0), *it), *it);
295  if (++i > formatting.range_length_limit) {
296  out = format_to(out, " ... <other elements>");
297  break;
298  }
299  }
300  if (formatting.add_prepostfix_space)
301  *out++ = ' ';
302  return internal::copy(formatting.postfix, out);
303  }
