144  FMT_CONSTEXPR const Char *on_format_specs(const Char *begin, const Char *end)
145  {
146  const auto specs_offset = to_unsigned(begin - format_.begin());
147 
148  using parse_context = basic_parse_context<Char>;
149  internal::dynamic_format_specs<Char> parsed_specs;
150  dynamic_specs_handler<parse_context> handler(parsed_specs, parse_context_);
151  begin = parse_format_specs(begin, end, handler);
152 
153  if (*begin != '}')
154  on_error("missing '}' in format string");
155 
156  auto &last_part = parts_.back();
157  auto specs = last_part.which == part::kind::argument_id
158  ? typename part::specification(last_part.val.arg_id)
159  : typename part::specification(last_part.val.named_arg_id);
160  specs.parsed_specs = parsed_specs;
161  last_part = part(specs);
162  last_part.end_of_argument_id = specs_offset;
163  return begin;
164  }
165 
166  private:
167  PartsContainer & parts_;
168  basic_string_view<Char> format_;
169  basic_parse_context<Char> parse_context_;

187  template <typename Range, typename Context>
188  auto vformat_to(Range out, basic_format_args<Context> args) const -> typename Context::iterator
189  {
190  const auto format_view = internal::to_string_view(format_);
191  basic_parse_context<char_type> parse_ctx(format_view);
192  Context ctx(out.begin(), args);
193 
194  const auto &parts = parts_provider_.parts();
195  for (auto part_it = parts.begin(); part_it != parts.end(); ++part_it) {
196  const auto &part = *part_it;
197  const auto &value = part.val;
198 
199  switch (part.which) {
200  case format_part_t::kind::text: {
201  const auto text = value.text.to_view(format_view.data());
202  auto output = ctx.out();
203  auto && it = internal::reserve(output, text.size());
204  it = std::copy_n(text.begin(), text.size(), it);
205  ctx.advance_to(output);
206  } break;
207 
208  case format_part_t::kind::argument_id: {
209  advance_parse_context_to_specification(parse_ctx, part);
210  format_arg<Range>(parse_ctx, ctx, value.arg_id);
211  } break;
212 
213  case format_part_t::kind::named_argument_id: {
214  advance_parse_context_to_specification(parse_ctx, part);
215  const auto named_arg_id = value.named_arg_id.to_view(format_view.data());
216  format_arg<Range>(parse_ctx, ctx, named_arg_id);
217  } break;
218  case format_part_t::kind::specification: {
219  const auto &arg_id_value = value.spec.arg_id.val;
220  const auto arg =
221  value.spec.arg_id.which == format_part_t::argument_id::which_arg_id::index
222  ? ctx.arg(arg_id_value.index)
223  : ctx.arg(arg_id_value.named_index.to_view(to_string_view(format_).data()));
224 
225  auto specs = value.spec.parsed_specs;
226 
227  handle_dynamic_spec<internal::width_checker>(specs.width, specs.width_ref, ctx,
228  format_view.begin());
229  handle_dynamic_spec<internal::precision_checker>(specs.precision, specs.precision_ref,
230  ctx, format_view.begin());
231 
232  check_prepared_specs(specs, arg.type());
233  advance_parse_context_to_specification(parse_ctx, part);
234  ctx.advance_to(visit_format_arg(arg_formatter<Range>(ctx, nullptr, &specs), arg));
235  } break;
236  }
237  }
238 
239  return ctx.out();
240  }
241 
242  private:
243  void advance_parse_context_to_specification(basic_parse_context<char_type> &parse_ctx,
244  const format_part_t & part) const
245  {
246  const auto view = to_string_view(format_);
247  const auto specification_begin = view.data() + part.end_of_argument_id;
248  advance_to(parse_ctx, specification_begin);
249  }

260  void check_prepared_specs(const basic_format_specs<Char> &specs, internal::type arg_type) const
261  {
262  internal::error_handler h;
263  numeric_specs_checker<internal::error_handler> checker(h, arg_type);
264  if (specs.align == align::numeric)
265  checker.require_numeric_argument();
266  if (specs.sign != sign::none)
267  checker.check_sign();

279  template <typename Char> struct part_counter
280  {
281  unsigned num_parts = 0;
282 
283  FMT_CONSTEXPR void on_text(const Char *begin, const Char *end)
284  {
285  if (begin != end)
286  ++num_parts;

290  FMT_CONSTEXPR void on_arg_id(unsigned) { ++num_parts; }
291  FMT_CONSTEXPR void on_arg_id(basic_string_view<Char>) { ++num_parts; }
292 
293  FMT_CONSTEXPR void on_replacement_field(const Char *) {}
294 
295  FMT_CONSTEXPR const Char *on_format_specs(const Char *begin, const Char *end)
296  {
297  // Find the matching brace.
298  unsigned braces_counter = 0;
299  for (; begin != end; ++begin) {
300  if (*begin == '{') {
301  ++braces_counter;
302  }
303  else if (*begin == '}') {
304  if (braces_counter == 0u)
305  break;
306  --braces_counter;
307  }
308  }
309  return begin;
310  }
